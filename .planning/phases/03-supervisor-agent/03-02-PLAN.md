**SupervisorAgent class:**

```python
class SupervisorAgent:
    """
    Supervisor agent for a single monitored inpatient.

    Runs a continuous evaluation cycle:
    1. Fetch/refresh FHIR data (parallel)
    2. Calculate deterministic clinical scores
    3. Generate score-based alerts
    4. LLM clinical reasoning
    5. Identify spawn triggers (Phase 6 hooks)

    The supervisor NEVER directly modifies treatment. All actions route
    through the approval queue.
    """

    def __init__(
        self,
        patient_id: str,
        encounter_id: str,
        model: str = DEFAULT_EVAL_MODEL,
        eval_interval_seconds: float = 60.0,
    ):
        self.state = PatientState(patient_id=patient_id, encounter_id=encounter_id)
        self.model = model
        self.eval_interval_seconds = eval_interval_seconds
        self._llm_client: OpenRouterClient | None = None
        self._system_prompt: str = ""
        self._running = False
        self._load_system_prompt()

    def _load_system_prompt(self):
        """Load the supervisor system prompt template."""
        if SUPERVISOR_PROMPT_PATH.exists():
            self._system_prompt = SUPERVISOR_PROMPT_PATH.read_text()
        else:
            logger.warning("Supervisor prompt not found at %s", SUPERVISOR_PROMPT_PATH)
            self._system_prompt = "You are a clinical supervisor agent. Analyze patient data and identify clinical concerns."

    def _get_llm_client(self) -> OpenRouterClient:
        """Lazy-initialize LLM client."""
        if self._llm_client is None:
            self._llm_client = OpenRouterClient(model=self.model)
        return self._llm_client

**Core evaluate() method (5 phases):**

```python
    async def evaluate(self) -> EvaluationResult:
        """Run one evaluation cycle for this patient."""
        cycle_id = str(uuid.uuid4())[:8]
        cycle_start = datetime.now(timezone.utc)
        logger.info("Supervisor eval cycle %s for patient %s", cycle_id, self.state.patient_id)

        # Phase 1: Fetch/refresh FHIR data (parallel)
        await self._refresh_patient_data()

        # Phase 2: Deterministic scoring (pure Python, <1ms)
        vitals_dict = self.state.get_vitals_dict()
        labs_dict = self.state.get_labs_dict()
        if vitals_dict:
            self.state.scores = calculate_all_available_scores(vitals_dict, labs_dict or None)
            logger.info("Scores calculated: %s", list(self.state.scores.keys()))
        else:
            logger.warning("No vitals available for scoring")
            self.state.scores = {}

        # Phase 3: Score-based alerting (deterministic)
        new_alerts = self._evaluate_score_alerts()

        # Phase 4: LLM clinical reasoning (async, ~5-15s)
        findings = await self._llm_evaluate()

        # Phase 5: Generate alerts from LLM findings + identify spawn triggers
        for finding in findings:
            alert = self._finding_to_alert(finding)
            if alert:
                new_alerts.append(alert.id)

        spawn_triggers = self._evaluate_spawn_triggers()

        # Identify data gaps
        data_gaps = self._identify_data_gaps(vitals_dict, labs_dict)

        cycle_duration = (datetime.now(timezone.utc) - cycle_start).total_seconds()
        logger.info("Eval cycle %s completed in %.1fs: %d alerts, %d findings",
                     cycle_id, cycle_duration, len(new_alerts), len(findings))

        result = EvaluationResult(
            cycle_id=cycle_id,
            timestamp=cycle_start,
            cycle_duration_seconds=cycle_duration,
            scores=self.state.scores,
            alerts_generated=new_alerts,
            findings=findings,
            spawn_triggers=spawn_triggers,
            data_gaps=data_gaps,
        )

        # Keep last 10 evaluations
        self.state.evaluation_history.append(result)
        if len(self.state.evaluation_history) > 10:
            self.state.evaluation_history = self.state.evaluation_history[-10:]

        self.state.last_updated = datetime.now(timezone.utc)
        return result
```

    async def _refresh_patient_data(self):
        """Fetch all FHIR data for the patient in parallel (NFR-01, NFR-05)."""
        tasks = []
        task_names = []

        if not self.state.is_vitals_fresh():
            tasks.append(self._fetch_vitals())
            task_names.append("vitals")
        if not self.state.is_conditions_fresh():
            tasks.append(self._fetch_conditions())
            task_names.append("conditions")
        if not self.state.is_meds_fresh():
            tasks.append(self._fetch_medications())
            task_names.append("medications")
        if not self.state.is_labs_fresh():
            tasks.append(self._fetch_labs())
            task_names.append("labs")

        # Also fetch patient name if not set
        if not self.state.patient_name:
            tasks.append(self._fetch_patient_info())
            task_names.append("patient_info")

        if not tasks:
            logger.debug("All data fresh, skipping FHIR fetch")
            return

        logger.info("Fetching FHIR data: %s", ", ".join(task_names))
        results = await asyncio.gather(*tasks, return_exceptions=True)
        for name, result in zip(task_names, results):
            if isinstance(result, Exception):
                logger.warning("FHIR fetch failed for %s: %s", name, result)

    async def _fetch_patient_info(self):
        """Fetch patient demographics."""
        try:
            result = await handle_get_patient({"patient_id": self.state.patient_id})

                if names:
                    name = names[0] if isinstance(names, list) else names
                    family = name.get("family", "")
                    given = " ".join(name.get("given", []))
                    self.state.patient_name = f"{given} {family}".strip()
        except Exception as e:
            logger.warning("Failed to fetch patient info: %s", e)

    def _parse_vitals(self, observations: list[dict]) -> list[VitalSigns]:
        """Parse FHIR Observation bundle into VitalSigns objects.

        Groups observations by effectiveDateTime (within 5-minute windows)
        to create composite VitalSigns records.
        """
        # Group by timestamp (round to 5-minute windows)
        time_groups: dict[str, list[dict]] = {}
        for obs in observations:
            effective = obs.get("effectiveDateTime", obs.get("issued", ""))
            if not effective:
                continue
            # Round to 5-minute window for grouping
            try:
                dt = datetime.fromisoformat(effective.replace("Z", "+00:00"))
                window_key = dt.strftime("%Y-%m-%dT%H:%M")  # minute precision
            except (ValueError, AttributeError):
                continue
            time_groups.setdefault(window_key, []).append(obs)

        vitals_list = []
        for window_key, obs_group in sorted(time_groups.items()):
            try:
                dt = datetime.fromisoformat(window_key + ":00+00:00")
            except ValueError:
                continue
            vs = VitalSigns(timestamp=dt)
            for obs in obs_group:
                self._apply_observation_to_vitals(obs, vs)
            vitals_list.append(vs)

        return vitals_list

    def _apply_observation_to_vitals(self, obs: dict, vs: VitalSigns):
        """Map a single FHIR Observation to VitalSigns fields."""
        obs_id = obs.get("id", "")
        if obs_id:
            vs.fhir_observation_ids.append(obs_id)

        # Get value - handle both valueQuantity and component-based observations
        code = ""
        codings = obs.get("code", {}).get("coding", [])
        if codings:
            code = codings[0].get("code", "")

        value = None
        if "valueQuantity" in obs:
            value = obs["valueQuantity"].get("value")
        elif "valueString" in obs:
            # Handle consciousness/AVPU
            value = obs.get("valueString")

        # Map LOINC codes to VitalSigns fields
        # Heart rate: 8867-4
        if code == "8867-4" and isinstance(value, (int, float)):
            vs.heart_rate = float(value)
        # Blood pressure: 85354-9 (panel with components)
        elif code == "85354-9":
            for comp in obs.get("component", []):
                comp_code = comp.get("code", {}).get("coding", [{}])[0].get("code", "")
                comp_value = comp.get("valueQuantity", {}).get("value")
                if comp_code == "8480-6" and comp_value is not None:  # Systolic
                    vs.systolic_bp = float(comp_value)
                elif comp_code == "8462-4" and comp_value is not None:  # Diastolic
                    vs.diastolic_bp = float(comp_value)
        # Respiratory rate: 9279-1
        elif code == "9279-1" and isinstance(value, (int, float)):
            vs.respiratory_rate = float(value)
        # SpO2: 2708-6 (or 59408-5)
        elif code in ("2708-6", "59408-5") and isinstance(value, (int, float)):
            vs.spo2 = float(value)
        # Temperature: 8310-5
        elif code == "8310-5" and isinstance(value, (int, float)):
            vs.temperature_c = float(value)
        # GCS: 9269-2
        elif code == "9269-2" and isinstance(value, (int, float)):
            vs.gcs = int(value)

    def _parse_labs(self, observations: list[dict]) -> list[LabResult]:
        """Parse FHIR Observation bundle into LabResult objects."""
        labs = []
        for obs in observations:
            codings = obs.get("code", {}).get("coding", [])
            if not codings:
                continue
            code = codings[0].get("code", "")
            display = codings[0].get("display", obs.get("code", {}).get("text", "Unknown"))

            value = None
            unit = ""
            if "valueQuantity" in obs:
                value = obs["valueQuantity"].get("value")
                unit = obs["valueQuantity"].get("unit", "")
            elif "valueString" in obs:
                # Try to parse numeric value from string
                try:
                    value = float(obs["valueString"])
                except (ValueError, TypeError):
                    continue

            if value is None:
                continue

            effective = obs.get("effectiveDateTime", obs.get("issued", ""))
            try:
                dt = datetime.fromisoformat(effective.replace("Z", "+00:00"))
            except (ValueError, AttributeError):
                dt = datetime.now(timezone.utc)

            # Check reference ranges
            ref_low = None
            ref_high = None
            is_abnormal = False
            for ref in obs.get("referenceRange", []):
                if "low" in ref:
                    ref_low = ref["low"].get("value")
                if "high" in ref:
                    ref_high = ref["high"].get("value")
            if ref_low is not None and value < ref_low:
                is_abnormal = True
            if ref_high is not None and value > ref_high:
                is_abnormal = True

            labs.append(LabResult(
                name=display,
                value=float(value),
                unit=unit,
                timestamp=dt,
                loinc_code=code,
                reference_low=ref_low,
                reference_high=ref_high,
                is_abnormal=is_abnormal,
                fhir_observation_id=obs.get("id", ""),
            ))

        return labs

**Score-based alert generation:**

```python
    def _evaluate_score_alerts(self) -> list[str]:
        """Generate alerts based on clinical score results (FR-05).

        Uses AlertClassifier for severity mapping and AlertManager for
        deduplication/suppression. Returns list of new alert IDs.
        """
        alert_ids = []
        mgr = get_alert_manager()

        # NEWS2-based alerts
        news2 = self.state.scores.get("NEWS2")
        if news2:
            any_3 = any(v == 3 for v in news2.get("scores", {}).values())
            severity = AlertClassifier.classify_from_news2(
                news2["total_score"], any_3
            )
            if severity in (AlertSeverity.CRITICAL, AlertSeverity.URGENT):
                alert = mgr.create_alert(
                    patient_id=self.state.patient_id,
                    severity=severity,
                    category="early_warning",
                    title=f"NEWS2 {news2['risk_level']}: Score {news2['total_score']}",
                    description=news2["clinical_response"],
                    source_agent="supervisor",
                    evidence=[{
                        "type": "clinical_score",
                        "name": "NEWS2",
                        "total_score": news2["total_score"],
                        "risk_level": news2["risk_level"],
                        "component_scores": news2.get("scores", {}),
                    }],
                    recommended_actions=[news2["clinical_response"]],
                )
                if alert:
                    alert_ids.append(alert.id)
                    self.state.active_alert_ids.append(alert.id)

        # qSOFA-based alerts
        qsofa = self.state.scores.get("qSOFA")
        if qsofa and qsofa.get("sepsis_risk"):
            severity = AlertClassifier.classify_from_qsofa(qsofa["score"])
            alert = mgr.create_alert(
                patient_id=self.state.patient_id,
                severity=severity,
                category="sepsis",
                title=f"qSOFA >= 2: Sepsis Risk",
                description=qsofa["recommendation"],
                source_agent="supervisor",
                evidence=[{
                    "type": "clinical_score",
                    "name": "qSOFA",
                    "score": qsofa["score"],
                    "criteria_met": qsofa.get("criteria_met", []),
                }],
                recommended_actions=[qsofa["recommendation"]],
            )
            if alert:
                alert_ids.append(alert.id)
                self.state.active_alert_ids.append(alert.id)

        # KDIGO-based alerts (AKI staging)
        kdigo = self.state.scores.get("KDIGO")
        if kdigo and kdigo.get("stage", 0) >= 2:
            severity = AlertSeverity.URGENT if kdigo["stage"] >= 3 else AlertSeverity.ROUTINE

    async def _llm_evaluate(self) -> list[Finding]:
        """Use LLM for clinical reasoning beyond deterministic scoring (NFR-04).

        Sends structured patient summary (not raw FHIR JSON) to the LLM
        with the supervisor system prompt. Returns parsed findings.
        """
        client = self._get_llm_client()

        # Build context
        clinical_summary = self.state.to_clinical_summary()
        scores_summary = self._format_scores_summary()

        # Fill template placeholders
        prompt = self._system_prompt
        prompt = prompt.replace("{patient_clinical_summary}", clinical_summary)
        prompt = prompt.replace("{scores_summary}", scores_summary)

        try:
            response = await client.create_message(
                messages=[{
                    "role": "user",
                    "content": (
                        "Evaluate the patient state provided in your system prompt. "
                        "Identify any clinical concerns, focusing on trends, cross-references between data, "
                        "and potential issues not captured by the deterministic scores. "
                        "Respond with a JSON array of findings."
                    ),
                }],
                system=prompt,
                tools=None,  # Supervisor eval doesn't use tools — data already in context
                max_tokens=2048,
                temperature=0.3,  # Lower temperature for clinical analysis
            )

            if response.content:
                return self._parse_llm_findings(response.content)
            return []

        except Exception as e:
            logger.error("LLM evaluation failed: %s", e)
            return []

    def _format_scores_summary(self) -> str:
        """Format clinical scores for LLM context."""
        if not self.state.scores:
            return "No clinical scores calculated (insufficient data)."
        lines = []
        for name, data in self.state.scores.items():
            if name == "NEWS2":
                lines.append(f"NEWS2: Total Score {data.get('total_score', 'N/A')} "
                           f"— Risk Level: {data.get('risk_level', 'N/A')} "
                           f"— Response: {data.get('clinical_response', 'N/A')}")
                scores = data.get("scores", {})
                if scores:
                    lines.append(f"  Component scores: {scores}")
            elif name == "qSOFA":
                lines.append(f"qSOFA: Score {data.get('score', 'N/A')} "
                           f"— Sepsis Risk: {data.get('sepsis_risk', False)} "
                           f"— Criteria Met: {data.get('criteria_met', [])}")
            elif name == "SOFA":
                lines.append(f"SOFA: Total Score {data.get('total_score', 'N/A')} "
                           f"— Component scores: {data.get('scores', {})}")
            elif name == "KDIGO":
                lines.append(f"KDIGO: Stage {data.get('stage', 'N/A')} "
                           f"— Criteria: {data.get('criteria', 'N/A')}")
            else:
                lines.append(f"{name}: {data}")
        return "\n".join(lines)

    def _parse_llm_findings(self, content: str) -> list[Finding]:
        """Parse LLM response into Finding objects.

        Handles both clean JSON and JSON embedded in markdown code blocks.
        """
        # Strip markdown code blocks if present
        text = content.strip()
        if text.startswith("```"):
            # Remove opening and closing code fences
            lines = text.split("\n")
            lines = [l for l in lines if not l.strip().startswith("```")]
            text = "\n".join(lines).strip()

        try:
            data = json.loads(text)
        except json.JSONDecodeError:
            # Try to extract JSON array from text
            start = text.find("[")
            end = text.rfind("]")
            if start >= 0 and end > start:
                try:
                    data = json.loads(text[start:end + 1])
                except json.JSONDecodeError:
                    logger.warning("Could not parse LLM findings as JSON")
                    return []
            else:
                logger.warning("No JSON array found in LLM response")
                return []

        if not isinstance(data, list):
            data = [data]

        findings = []
        for item in data:
            if not isinstance(item, dict):
                continue
            try:
                findings.append(Finding(
                    category=item.get("category", "general"),
                    severity=item.get("severity", "informational"),
                    title=item.get("title", ""),
                    description=item.get("description", ""),
                    evidence=item.get("evidence", []),
                    recommended_actions=item.get("recommended_actions", []),
                    spawn_trigger=item.get("spawn_trigger"),
                ))
            except Exception as e:
                logger.warning("Failed to parse finding: %s", e)
                continue

        return findings
```

    def _finding_to_alert(self, finding: Finding):
        """Convert an LLM Finding to an alert via AlertManager.

        Only converts findings with severity >= ROUTINE.
        Returns the Alert or None if suppressed/deduplicated.
        """
        severity_map = {
            "critical": AlertSeverity.CRITICAL,
            "urgent": AlertSeverity.URGENT,
            "routine": AlertSeverity.ROUTINE,
            "informational": AlertSeverity.INFORMATIONAL,
        }
        severity = severity_map.get(finding.severity, AlertSeverity.INFORMATIONAL)

        # Only generate alerts for ROUTINE and above — INFORMATIONAL findings are logged but not alerted
        if severity == AlertSeverity.INFORMATIONAL:
            return None

        mgr = get_alert_manager()
        return mgr.create_alert(
            patient_id=self.state.patient_id,
            severity=severity,
            category=finding.category,
            title=finding.title,
            description=finding.description,
            source_agent="supervisor_llm",
            evidence=finding.evidence,
            recommended_actions=finding.recommended_actions,
        )

    def _evaluate_spawn_triggers(self) -> list[str]:
        """Identify conditions warranting specialist agents (Phase 6 hooks).

        Based on score results, active conditions, and LLM findings.
        """
        triggers = set()

        # Score-based triggers
        qsofa = self.state.scores.get("qSOFA", {})
        if qsofa.get("sepsis_risk"):
            triggers.add("infectious_disease")

        kdigo = self.state.scores.get("KDIGO", {})
        if kdigo.get("stage", 0) >= 1:
            triggers.add("renal")

        # Condition-based triggers
        for condition in self.state.active_conditions:
            code = ""
            codings = condition.get("code", {}).get("coding", [])
            if codings:
                code = codings[0].get("code", "")

            if code in CARDIAC_SNOMED_CODES:
                triggers.add("cardiology")
            if code in SEPSIS_SNOMED_CODES:
                triggers.add("infectious_disease")
            if code in RENAL_SNOMED_CODES:
                triggers.add("renal")
            if code in PULMONARY_SNOMED_CODES:
                triggers.add("pulmonary")

        # LLM finding-based triggers
        for result in self.state.evaluation_history[-1:]:
            for finding in result.findings:
                if finding.spawn_trigger:
                    triggers.add(finding.spawn_trigger)

        self.state.spawn_triggers = sorted(triggers)
        return self.state.spawn_triggers

    def _identify_data_gaps(self, vitals_dict: dict, labs_dict: dict) -> list[str]:
        """Identify missing data that could affect clinical decisions."""
        gaps = []

        # Check for missing vital signs
        if not vitals_dict:
            gaps.append("No vital signs available")
        else:
            expected_vitals = ["heart_rate", "systolic_bp", "respiratory_rate", "spo2", "temperature_c"]
            for v in expected_vitals:
                if v not in vitals_dict:
                    gaps.append(f"Missing vital sign: {v}")
            if "gcs" not in vitals_dict:
                gaps.append("Missing GCS for qSOFA calculation")

        # Check for missing labs relevant to scoring
        if not labs_dict:
            gaps.append("No laboratory results available")
        else:
            if "creatinine" not in labs_dict:
                gaps.append("Missing creatinine for KDIGO/SOFA scoring")
            if "platelets" not in labs_dict:
                gaps.append("Missing platelets for SOFA scoring")

        return gaps

```python
    async def start_monitoring(self):
        """Start the continuous monitoring loop.

        Runs evaluate() at the configured interval until stopped.
        """
        self._running = True
        logger.info("Starting supervisor monitoring for patient %s (interval: %.0fs)",
                    self.state.patient_id, self.eval_interval_seconds)

        while self._running:
            try:
                result = await self.evaluate()
                # Adjust interval based on acuity
                self._adjust_eval_interval(result)
            except Exception as e:
                logger.error("Evaluation cycle failed for patient %s: %s",
                           self.state.patient_id, e)

            if self._running:
                await asyncio.sleep(self.eval_interval_seconds)

        logger.info("Supervisor monitoring stopped for patient %s", self.state.patient_id)

    def stop_monitoring(self):
        """Stop the continuous monitoring loop."""
        self._running = False
        logger.info("Stop requested for supervisor of patient %s", self.state.patient_id)

    def _adjust_eval_interval(self, result: EvaluationResult):
        """Adjust evaluation interval based on patient acuity.

        Critical patients: 30s, Urgent: 60s, Routine/Stable: 120s
        """
        news2 = result.scores.get("NEWS2", {})
        risk = news2.get("risk_level", "LOW")

        if risk == "HIGH":
            self.eval_interval_seconds = 30.0
        elif risk in ("MEDIUM", "LOW_KEY_CONCERN"):
            self.eval_interval_seconds = 60.0
        else:
            self.eval_interval_seconds = 120.0

    async def close(self):
        """Clean up resources."""
        self.stop_monitoring()
        if self._llm_client:
            await self._llm_client.close()
