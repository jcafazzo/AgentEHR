**SupervisorManager class:**

```python
class SupervisorManager:
    """
    Manages active SupervisorAgent instances.

    Enforces one-supervisor-per-patient constraint (FR-02).
    Tracks background monitoring tasks for lifecycle control.
    """

    def __init__(self):
        self._supervisors: dict[str, SupervisorAgent] = {}  # patient_id -> agent
        self._tasks: dict[str, asyncio.Task] = {}  # patient_id -> monitoring task
        logger.info("SupervisorManager initialized")

    async def start_supervisor(
        self,
        patient_id: str,
        encounter_id: str,
        model: str = "gemini-3-flash",
        eval_interval_seconds: float = 60.0,
    ) -> dict[str, Any]:
        """Start a supervisor for a patient.

        If a supervisor already exists for this patient, returns its current state
        without creating a duplicate (FR-02 one-per-patient).

        Returns dict with status information.
        """
        if patient_id in self._supervisors:
            agent = self._supervisors[patient_id]
            logger.info("Supervisor already active for patient %s", patient_id)
            return {
                "status": "already_active",
                "patient_id": patient_id,
                "encounter_id": agent.state.encounter_id,
                "last_updated": agent.state.last_updated.isoformat() if agent.state.last_updated else None,
                "eval_count": len(agent.state.evaluation_history),
            }

        # Create new supervisor
        agent = SupervisorAgent(
            patient_id=patient_id,
            encounter_id=encounter_id,
            model=model,
            eval_interval_seconds=eval_interval_seconds,
        )
        self._supervisors[patient_id] = agent

        # Run first evaluation immediately, then start background loop
        try:
            result = await agent.evaluate()
            logger.info("Initial evaluation for patient %s: %d alerts, %d findings",
                       patient_id, len(result.alerts_generated), len(result.findings))
        except Exception as e:
            logger.error("Initial evaluation failed for patient %s: %s", patient_id, e)

        # Start background monitoring task
        task = asyncio.create_task(agent.start_monitoring())
        self._tasks[patient_id] = task
        task.add_done_callback(lambda t: self._on_task_done(patient_id, t))

        return {
            "status": "started",
            "patient_id": patient_id,
            "encounter_id": encounter_id,
            "model": model,
            "eval_interval_seconds": eval_interval_seconds,
        }

    async def stop_supervisor(self, patient_id: str) -> dict[str, Any]:
        """Stop monitoring for a patient."""
        if patient_id not in self._supervisors:
            return {"status": "not_found", "patient_id": patient_id}

        agent = self._supervisors[patient_id]
        agent.stop_monitoring()

        # Cancel the background task
        task = self._tasks.pop(patient_id, None)
        if task and not task.done():
            task.cancel()
            try:
                await asyncio.wait_for(asyncio.shield(task), timeout=5.0)
            except (asyncio.CancelledError, asyncio.TimeoutError):
                pass

        # Clean up
        await agent.close()
        del self._supervisors[patient_id]

        logger.info("Supervisor stopped for patient %s", patient_id)
        return {"status": "stopped", "patient_id": patient_id}

    def get_supervisor(self, patient_id: str) -> SupervisorAgent | None:
        """Get the active supervisor for a patient."""
        return self._supervisors.get(patient_id)

    def get_state(self, patient_id: str) -> dict[str, Any] | None:
        """Get the current state for a supervised patient.

        Returns a serializable dict with clinical state, scores, and alerts.
        """
        agent = self._supervisors.get(patient_id)
        if not agent:
            return None

        state = agent.state
        latest_vitals = state.latest_vitals()

        return {
            "patient_id": state.patient_id,
            "encounter_id": state.encounter_id,
            "patient_name": state.patient_name,
            "last_updated": state.last_updated.isoformat() if state.last_updated else None,
            "is_monitoring": agent._running,
            "eval_interval_seconds": agent.eval_interval_seconds,
            "scores": state.scores,
            "active_alert_ids": state.active_alert_ids,
            "spawn_triggers": state.spawn_triggers,
            "data_gaps": state.evaluation_history[-1].data_gaps if state.evaluation_history else [],
            "latest_vitals": {
                "heart_rate": latest_vitals.heart_rate if latest_vitals else None,
                "systolic_bp": latest_vitals.systolic_bp if latest_vitals else None,
                "diastolic_bp": latest_vitals.diastolic_bp if latest_vitals else None,
                "respiratory_rate": latest_vitals.respiratory_rate if latest_vitals else None,
                "spo2": latest_vitals.spo2 if latest_vitals else None,
                "temperature_c": latest_vitals.temperature_c if latest_vitals else None,
                "timestamp": latest_vitals.timestamp.isoformat() if latest_vitals else None,
            } if latest_vitals else None,
            "conditions_count": len(state.active_conditions),
            "medications_count": len(state.active_medications),
            "eval_count": len(state.evaluation_history),
            "last_eval": {
                "cycle_id": state.evaluation_history[-1].cycle_id,
                "duration_seconds": state.evaluation_history[-1].cycle_duration_seconds,
                "alerts_count": len(state.evaluation_history[-1].alerts_generated),
                "findings_count": len(state.evaluation_history[-1].findings),
            } if state.evaluation_history else None,
            "clinical_summary": state.to_clinical_summary(),
        }

    def list_supervisors(self) -> list[dict[str, Any]]:
        """List all active supervisors with summary information."""
        result = []
        for patient_id, agent in self._supervisors.items():
            state = agent.state
            news2 = state.scores.get("NEWS2", {})
            result.append({
                "patient_id": patient_id,
                "encounter_id": state.encounter_id,
                "patient_name": state.patient_name,
                "is_monitoring": agent._running,
                "last_updated": state.last_updated.isoformat() if state.last_updated else None,
                "news2_score": news2.get("total_score"),
                "news2_risk": news2.get("risk_level"),
                "active_alerts": len(state.active_alert_ids),
                "spawn_triggers": state.spawn_triggers,
            })
        # Sort by NEWS2 score descending (most acute first)
        result.sort(key=lambda x: x.get("news2_score") or 0, reverse=True)
        return result

    async def stop_all(self):
        """Stop all active supervisors. Used during server shutdown."""
        patient_ids = list(self._supervisors.keys())
        for pid in patient_ids:
            await self.stop_supervisor(pid)
        logger.info("All supervisors stopped")

    def _on_task_done(self, patient_id: str, task: asyncio.Task):
        """Callback when a monitoring task completes (normally or via error)."""
        if task.cancelled():
            logger.info("Monitoring task cancelled for patient %s", patient_id)
        elif task.exception():
            logger.error("Monitoring task failed for patient %s: %s",
                        patient_id, task.exception())
        else:
            logger.info("Monitoring task completed for patient %s", patient_id)
```
