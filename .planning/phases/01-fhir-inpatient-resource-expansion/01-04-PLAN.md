#!/usr/bin/env python3
"""
Integration test for inpatient FHIR handlers.

Tests the complete inpatient encounter lifecycle with all resource types.
Requires a running Medplum instance with seeded patient data.

Run from the fhir-mcp-server directory:
  python test_inpatient_handlers.py
"""

import asyncio
import sys
import os

sys.path.insert(0, os.path.join(os.path.dirname(__file__), "src"))

# Import all handlers from handlers.py (not server.py)
from handlers import (
    # Encounter lifecycle

**Test flow (single async function `test_full_inpatient_encounter()`):**

1. **Find a test patient** -- Search for any patient. Use first result's ID. If no patients found, print error and return False.

2. **Create inpatient encounter** -- Call `handle_create_inpatient_encounter` with patient_id, reason="Sepsis workup", admit_source="emd", location="Emergency Department", priority="urgent". Assert result has no "error" key and has an encounter id (check for key in "encounter" sub-dict or directly). Store encounter_id.

3. **Create care team** -- Call `handle_create_care_team` with patient_id, name=f"ICU Care Team", encounter_id, participants=[{"role_code": "309343006", "role_display": "Physician", "member_display": "Dr. Williams"}, {"role_code": "224535009", "role_display": "Registered nurse", "member_display": "RN Johnson"}]. Assert no error. Store care_team_id.

4. **Create clinical flag** -- Call `handle_create_flag` with patient_id, description="Sepsis alert - qSOFA >= 2", category="clinical", encounter_id, priority="PH". Assert no error. Store flag_id.

5. **Create clinical impression** -- Call `handle_create_clinical_impression` with patient_id, encounter_id, summary="Patient meets qSOFA >= 2. Suspected sepsis.", assessor="Infectious Disease Agent", finding_code="91302008", finding_display="Sepsis". Assert no error.

6. **Create risk assessment** -- Call `handle_create_risk_assessment` with patient_id, encounter_id, condition_display="Septic shock", outcome_text="Deterioration within 6 hours", risk_level="high", basis=["NEWS2 score: 9", "qSOFA score: 2"]. Assert no error.

7. **Create tasks** -- Call `handle_create_task` with patient_id, encounter_id, description="Draw blood cultures", priority="stat", requester="Supervisor Agent", owner="RN Johnson", notes=["Part of Hour-1 Bundle"]. Assert no error. Store task_id.

8. **Assign and complete task** -- Call `handle_assign_task` with task_id, owner="RN Johnson". Then call `handle_complete_task` with task_id, output_text="Blood cultures drawn from 2 sites". Assert no error on both.

9. **Create goal** -- Call `handle_create_goal` with patient_id, description="MAP > 65 mmHg", encounter_id, target_measure_code="8478-0", target_measure_display="Mean blood pressure", target_value=65, target_unit="mmHg". Assert no error. Store goal_id.

10. **Transfer patient** -- Call `handle_transfer_patient` with encounter_id, new_location="ICU Bed 3". Assert no error.

11. **Record device metric** -- Call `handle_record_device_metric` with type_code="150456", type_display="SpO2", source_display="Bedside Monitor - ICU Bed 3". Assert no error.

12. **Report adverse event** -- Call `handle_report_adverse_event` with patient_id, encounter_id, event_description="Mild allergic reaction to initial antibiotic", category_code="medication-mishap", seriousness="non-serious", severity="mild". Assert no error.

13. **Create communication** -- Call `handle_create_inpatient_communication` with patient_id, encounter_id, content="Handoff: Patient admitted with sepsis. Blood cultures pending. On empiric antibiotics. MAP target 65.", category="handoff", sender="Dr. Williams", recipient="Dr. Night Shift". Assert no error.

14. **Get encounter timeline** -- Call `handle_get_encounter_timeline` with encounter_id. Assert result contains encounter info with status and location.

15. **Verify reads** -- Call each search/get handler to verify resources are retrievable:
    - `handle_get_active_flags(patient_id, encounter_id)` -- assert total >= 1
    - `handle_get_clinical_impressions(patient_id, encounter_id)` -- assert total >= 1
    - `handle_get_risk_assessments(patient_id, encounter_id)` -- assert total >= 1
    - `handle_get_pending_tasks(patient_id, encounter_id)` -- check total (may be 0 since task was completed)
    - `handle_get_care_team(patient_id, encounter_id)` -- assert total >= 1
    - `handle_get_patient_goals(patient_id)` -- assert total >= 1
    - `handle_get_device_metrics()` -- assert total >= 1
    - `handle_get_adverse_events(patient_id, encounter_id)` -- assert total >= 1
    - `handle_get_communications(encounter_id)` -- assert total >= 1
    - `handle_search_communications(patient_id, encounter_id=encounter_id)` -- assert total >= 1

16. **Resolve flag** -- Call `handle_resolve_flag` with flag_id. Assert no error and status is "inactive".

17. **Update goal** -- Call `handle_update_goal_status` with goal_id, lifecycle_status="completed". Assert no error.

18. **Discharge patient** -- Call `handle_discharge_patient` with encounter_id, discharge_disposition="home". Assert no error and status is "finished".

**Each step should:**
- Print a descriptive message (e.g. "Step 1: Finding test patient...")
- Print the key result fields
- If any step returns an "error" key, print the error and return False
- Track pass/fail for summary
